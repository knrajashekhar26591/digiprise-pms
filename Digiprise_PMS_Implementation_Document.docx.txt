public class Issue : BaseEntity
{
    public int Id             { get; private set; }
    public int TenantId       { get; private set; }
    public int ProjectId      { get; private set; }
    public string IssueKey    { get; private set; } = default!; // e.g. DIG-42
    public IssueType IssueType { get; private set; }
    public string Summary     { get; private set; } = default!;
    public string? Description { get; private set; }
    public int StatusId       { get; private set; }
    public Priority Priority  { get; private set; }
    public int? AssigneeId    { get; private set; }
    public int ReporterId     { get; private set; }
    public int? ParentIssueId { get; private set; }
    public int? SprintId      { get; private set; }
    public int? StoryPoints   { get; private set; }
    public int? EpicId        { get; private set; }
    public DateTime CreatedAt { get; private set; }
    public DateTime UpdatedAt { get; private set; }
    public DateTime? DueDate  { get; private set; }
 
    // Navigation
    public Project Project          { get; private set; } = default!;
    public WorkflowStatus Status    { get; private set; } = default!;
    public User? Assignee           { get; private set; }
    public IReadOnlyList&lt;Comment&gt; Comments    =&gt; _comments.AsReadOnly();
    public IReadOnlyList&lt;IssueLink&gt; Links     =&gt; _links.AsReadOnly();
    public IReadOnlyList&lt;Attachment&gt; Attachments =&gt; _attachments.AsReadOnly();
 
    private readonly List&lt;Comment&gt;    _comments    = [];
    private readonly List&lt;IssueLink&gt;  _links       = [];
    private readonly List&lt;Attachment&gt; _attachments = [];
 
    // Domain events collection
    private readonly List&lt;IDomainEvent&gt; _domainEvents = [];
    public IReadOnlyList&lt;IDomainEvent&gt; DomainEvents =&gt; _domainEvents.AsReadOnly();
 
    // Factory
    public static Issue Create(int tenantId, int projectId, string key,
        IssueType type, string summary, Priority priority, int reporterId)
    {
        var issue = new Issue
        {
            TenantId   = tenantId,
            ProjectId  = projectId,
            IssueKey   = key,
            IssueType  = type,
            Summary    = summary,
            Priority   = priority,
            ReporterId = reporterId,
            CreatedAt  = DateTime.UtcNow,
            UpdatedAt  = DateTime.UtcNow
        };
        issue._domainEvents.Add(new IssueCreatedEvent(issue));
        return issue;
    }
 
    public void Transition(WorkflowStatus newStatus, int actorId)
    {
        var old = StatusId;
        StatusId  = newStatus.Id;
        UpdatedAt = DateTime.UtcNow;
        _domainEvents.Add(new IssueStatusChangedEvent(Id, old, newStatus.Id, actorId));
    }
}

// Infrastructure/Persistence/PmsDbContext.cs
public class PmsDbContext : DbContext
{
    private readonly ITenantContext _tenant;
 
    public PmsDbContext(DbContextOptions&lt;PmsDbContext&gt; opts, ITenantContext tenant)
        : base(opts) =&gt; _tenant = tenant;
 
    public DbSet&lt;Project&gt;          Projects          =&gt; Set&lt;Project&gt;();
    public DbSet&lt;Issue&gt;            Issues            =&gt; Set&lt;Issue&gt;();
    public DbSet&lt;Sprint&gt;           Sprints           =&gt; Set&lt;Sprint&gt;();
    public DbSet&lt;Comment&gt;          Comments          =&gt; Set&lt;Comment&gt;();
    public DbSet&lt;Attachment&gt;       Attachments       =&gt; Set&lt;Attachment&gt;();
    public DbSet&lt;WorkflowStatus&gt;   WorkflowStatuses  =&gt; Set&lt;WorkflowStatus&gt;();
    public DbSet&lt;IssueHistory&gt;     IssueHistories    =&gt; Set&lt;IssueHistory&gt;();
    public DbSet&lt;Notification&gt;     Notifications     =&gt; Set&lt;Notification&gt;();
    public DbSet&lt;AutomationRule&gt;   AutomationRules   =&gt; Set&lt;AutomationRule&gt;();
    public DbSet&lt;SprintSnapshot&gt;   SprintSnapshots   =&gt; Set&lt;SprintSnapshot&gt;();
    public DbSet&lt;AuditLog&gt;         AuditLogs         =&gt; Set&lt;AuditLog&gt;();
 
    protected override void OnModelCreating(ModelBuilder mb)
    {
        mb.ApplyConfigurationsFromAssembly(typeof(PmsDbContext).Assembly);
 
        // ── Global tenant filter (Row-Level Security) ──
        mb.Entity&lt;Issue&gt;().HasQueryFilter(i =&gt; i.TenantId == _tenant.Id);
        mb.Entity&lt;Project&gt;().HasQueryFilter(p =&gt; p.TenantId == _tenant.Id);
        mb.Entity&lt;Sprint&gt;().HasQueryFilter(s =&gt; s.TenantId == _tenant.Id);
        // ... apply to all tenant-scoped tables
    }
}

// Infrastructure/Persistence/Configurations/IssueConfiguration.cs
public class IssueConfiguration : IEntityTypeConfiguration&lt;Issue&gt;
{
    public void Configure(EntityTypeBuilder&lt;Issue&gt; b)
    {
        b.ToTable(&quot;Issues&quot;);
        b.HasKey(i =&gt; i.Id);
 
        b.Property(i =&gt; i.IssueKey).HasMaxLength(20).IsRequired();
        b.Property(i =&gt; i.Summary).HasMaxLength(255).IsRequired();
        b.Property(i =&gt; i.Description).HasColumnType(&quot;nvarchar(max)&quot;);
        b.Property(i =&gt; i.IssueType).HasConversion&lt;string&gt;().HasMaxLength(30);
        b.Property(i =&gt; i.Priority).HasConversion&lt;string&gt;().HasMaxLength(20);
 
        // ── Indexes ──
        b.HasIndex(i =&gt; new { i.TenantId, i.ProjectId, i.StatusId })  // board queries
            .HasDatabaseName(&quot;IX_Issues_Tenant_Project_Status&quot;);
        b.HasIndex(i =&gt; new { i.TenantId, i.AssigneeId, i.UpdatedAt })  // assigned-to-me
            .HasDatabaseName(&quot;IX_Issues_Tenant_Assignee_Updated&quot;);
        b.HasIndex(i =&gt; i.IssueKey).IsUnique()
            .HasDatabaseName(&quot;UQ_Issues_IssueKey&quot;);
 
        // ── Relationships ──
        b.HasMany(i =&gt; i.Comments).WithOne().HasForeignKey(c =&gt; c.IssueId);
        b.HasMany(i =&gt; i.Attachments).WithOne().HasForeignKey(a =&gt; a.IssueId);
    }
}

[ApiController]
[Route(&quot;api/v1/[controller]&quot;)]
[Authorize]
public class IssuesController : ControllerBase
{
    private readonly IMediator _mediator;
    public IssuesController(IMediator mediator) =&gt; _mediator = mediator;
 
    // GET /api/v1/issues/{id}
    [HttpGet(&quot;{id:int}&quot;)]
    [ProducesResponseType&lt;IssueDetailDto&gt;(200)]
    [ProducesResponseType(404)]
    public async Task&lt;IActionResult&gt; GetById(int id, CancellationToken ct)
    {
        var result = await _mediator.Send(new GetIssueByIdQuery(id), ct);
        return result is null ? NotFound() : Ok(result);
    }
 
    // POST /api/v1/issues
    [HttpPost]
    [ProducesResponseType&lt;IssueDto&gt;(201)]
    public async Task&lt;IActionResult&gt; Create([FromBody] CreateIssueRequest req, CancellationToken ct)
    {
        var cmd = new CreateIssueCommand(
            req.ProjectKey, req.IssueType, req.Summary,
            req.Description, req.Priority, req.AssigneeId,
            req.StoryPoints, req.SprintId, req.Labels, req.CustomFields);
 
        var result = await _mediator.Send(cmd, ct);
        return CreatedAtAction(nameof(GetById), new { id = result.Id }, result);
    }
 
    // PATCH /api/v1/issues/{id}
    [HttpPatch(&quot;{id:int}&quot;)]
    public async Task&lt;IActionResult&gt; Update(int id,
        [FromBody] UpdateIssueRequest req, CancellationToken ct)
    {
        var cmd = new UpdateIssueCommand(id, req);
        await _mediator.Send(cmd, ct);
        return NoContent();
    }
 
    // POST /api/v1/issues/search  (IQL)
    [HttpPost(&quot;search&quot;)]
    [ProducesResponseType&lt;PagedResult&lt;IssueDto&gt;&gt;(200)]
    public async Task&lt;IActionResult&gt; Search([FromBody] IssueSearchRequest req, CancellationToken ct)
    {

            new SearchIssuesQuery(req.Iql, req.MaxResults, req.StartAt), ct);

// Application/Issues/Commands/CreateIssueCommand.cs
public record CreateIssueCommand(
    string ProjectKey,
    string IssueType,
    string Summary,
    string? Description,
    string Priority,
    int? AssigneeId,
    int? StoryPoints,
    int? SprintId,
    List&lt;string&gt; Labels,
    Dictionary&lt;string, object&gt; CustomFields
) : IRequest&lt;IssueDto&gt;;
 
// Application/Issues/Commands/CreateIssueCommandHandler.cs
public class CreateIssueCommandHandler : IRequestHandler&lt;CreateIssueCommand, IssueDto&gt;
{
    private readonly PmsDbContext _db;
    private readonly ITenantContext _tenant;
    private readonly IPublisher _eventBus;
    private readonly IMapper _mapper;
    private readonly IIssueKeyGenerator _keyGen;
 
    public CreateIssueCommandHandler(PmsDbContext db, ITenantContext tenant,
        IPublisher eventBus, IMapper mapper, IIssueKeyGenerator keyGen)
    {
        _db = db; _tenant = tenant;
        _eventBus = eventBus; _mapper = mapper; _keyGen = keyGen;
    }
 
    public async Task&lt;IssueDto&gt; Handle(CreateIssueCommand cmd, CancellationToken ct)
    {
        var project = await _db.Projects
            .FirstOrDefaultAsync(p =&gt; p.Key == cmd.ProjectKey, ct)
            ?? throw new NotFoundException($&quot;Project {cmd.ProjectKey} not found&quot;);
 
        var key = await _keyGen.NextKeyAsync(project.Id, ct);
 
        var issue = Issue.Create(
            _tenant.Id, project.Id, key,
            Enum.Parse&lt;IssueType&gt;(cmd.IssueType),
            cmd.Summary,
            Enum.Parse&lt;Priority&gt;(cmd.Priority),
            _tenant.UserId);
 
        issue.SetDescription(cmd.Description);
        if (cmd.AssigneeId.HasValue) issue.Assign(cmd.AssigneeId.Value);
        if (cmd.StoryPoints.HasValue) issue.SetStoryPoints(cmd.StoryPoints.Value);
        if (cmd.SprintId.HasValue) issue.AddToSprint(cmd.SprintId.Value);
 
        _db.Issues.Add(issue);
        await _db.SaveChangesAsync(ct);
 
        // Publish domain events
        foreach (var evt in issue.DomainEvents)
            await _eventBus.Publish(evt, ct);
 
        return _mapper.Map&lt;IssueDto&gt;(issue);
    }
}
 
3.3 FluentValidation – CreateIssueValidator
 
// Application/Issues/Commands/CreateIssueCommandValidator.cs
public class CreateIssueCommandValidator : AbstractValidator&lt;CreateIssueCommand&gt;
{
    public CreateIssueCommandValidator()
    {
        RuleFor(x =&gt; x.ProjectKey)
            .NotEmpty()
            .MaximumLength(10)
            .Matches(@&quot;^[A-Z]{2,10}$&quot;)
            .WithMessage(&quot;Project key must be 2-10 uppercase letters&quot;);
 
        RuleFor(x =&gt; x.Summary)
            .NotEmpty()
            .MaximumLength(255)
            .WithMessage(&quot;Summary is required and max 255 characters&quot;);
 
        RuleFor(x =&gt; x.IssueType)
            .NotEmpty()
            .Must(t =&gt; Enum.TryParse&lt;IssueType&gt;(t, out _))
            .WithMessage(&quot;Invalid issue type&quot;);
 
        RuleFor(x =&gt; x.Priority)
            .NotEmpty()
            .Must(p =&gt; Enum.TryParse&lt;Priority&gt;(p, out _))
            .WithMessage(&quot;Invalid priority&quot;);
 
        RuleFor(x =&gt; x.StoryPoints)
            .GreaterThan(0)
            .LessThanOrEqualTo(999)
            .When(x =&gt; x.StoryPoints.HasValue);
    }
}
 
3.4 JWT Authentication Setup
 
// Extensions/ServiceCollectionExtensions.cs
public static IServiceCollection AddPmsAuthentication(
    this IServiceCollection services, IConfiguration config)
{
    var jwtSettings = config.GetSection(&quot;Jwt&quot;).Get&lt;JwtSettings&gt;()!;
 
    services.AddAuthentication(JwtBearerDefaults.AuthenticationScheme)
        .AddJwtBearer(opts =&gt;
        {
            opts.TokenValidationParameters = new TokenValidationParameters
            {
                ValidateIssuer           = true,
                ValidateAudience         = true,
                ValidateLifetime         = true,
                ValidateIssuerSigningKey = true,
                ValidIssuer              = jwtSettings.Issuer,
                ValidAudience            = jwtSettings.Audience,
                IssuerSigningKey         = new SymmetricSecurityKey(
                    Encoding.UTF8.GetBytes(jwtSettings.SecretKey)),
                ClockSkew                = TimeSpan.FromSeconds(30)
            };
            // Support SignalR JWT in query string
            opts.Events = new JwtBearerEvents
            {
                OnMessageReceived = ctx =&gt;
                {
                    var token = ctx.Request.Query[&quot;access_token&quot;];
                    if (!string.IsNullOrEmpty(token) &amp;&amp;
                        ctx.HttpContext.Request.Path.StartsWithSegments(&quot;/hubs&quot;))
                        ctx.Token = token;
                    return Task.CompletedTask;
                }
            };
        });
 
    services.AddAuthorization(opts =&gt;
    {
        opts.AddPolicy(Policies.CreateIssue, p =&gt; p.RequireClaim(&quot;permission&quot;, &quot;issues.create&quot;));
        opts.AddPolicy(Policies.DeleteIssue, p =&gt; p.RequireClaim(&quot;permission&quot;, &quot;issues.delete&quot;));
        opts.AddPolicy(Policies.ManageWorkflow, p =&gt; p.RequireClaim(&quot;permission&quot;, &quot;workflow.manage&quot;));
    });
 
    return services;
}

// Application/Workflows/WorkflowEngine.cs
public class WorkflowEngine : IWorkflowEngine
{
    private readonly PmsDbContext _db;
    private readonly IValidatorFactory _validators;
    private readonly IPostFunctionExecutor _postFunctions;
 
    public WorkflowEngine(PmsDbContext db, IValidatorFactory validators,
        IPostFunctionExecutor postFunctions)
    {
        _db = db; _validators = validators; _postFunctions = postFunctions;
    }
 
    public async Task&lt;TransitionResult&gt; TransitionAsync(
        Issue issue, int transitionId, User actor, CancellationToken ct)
    {
        var transition = await _db.WorkflowTransitions
            .Include(t =&gt; t.ToStatus)
            .FirstOrDefaultAsync(t =&gt; t.Id == transitionId &amp;&amp;
                t.FromStatusId == issue.StatusId, ct)
            ?? throw new InvalidTransitionException(transitionId);
 
        // 1. Evaluate conditions
        var conditionOk = await EvaluateConditionsAsync(
            transition.Conditions, issue, actor, ct);
        if (!conditionOk)
            return TransitionResult.Forbidden(&quot;Transition condition not met&quot;);
 
        // 2. Run validators
        foreach (var validator in transition.Validators)
        {
            var vResult = await _validators
                .Create(validator.Type)
                .ValidateAsync(issue, validator.Config, ct);
            if (!vResult.IsValid)
                return TransitionResult.Failure(vResult.Error);
        }
 
        // 3. Execute transition
        issue.Transition(transition.ToStatus, actor.Id);
        await _db.SaveChangesAsync(ct);
 
        // 4. Run post-functions
        foreach (var pf in transition.PostFunctions)
            await _postFunctions.ExecuteAsync(pf, issue, actor, ct);
 
        return TransitionResult.Success();
    }
 
    private async Task&lt;bool&gt; EvaluateConditionsAsync(
        IEnumerable&lt;WorkflowCondition&gt; conditions,
        Issue issue, User actor, CancellationToken ct)
    {
        foreach (var c in conditions)
        {
            bool ok = c.Type switch
            {
                &quot;UserInRole&quot;  =&gt; actor.Roles.Contains(c.Config[&quot;role&quot;]),
                &quot;FieldNotEmpty&quot; =&gt; !string.IsNullOrEmpty(
                    issue.GetFieldValue(c.Config[&quot;field&quot;])),
                _ =&gt; true
            };
            if (!ok) return false;
        }
        return true;
    }
}
 
4.2 Automation Rule Engine
 
// Application/Automation/AutomationRuleProcessor.cs
public class AutomationRuleProcessor : IAutomationRuleProcessor
{
    private readonly PmsDbContext _db;
    private readonly IActionExecutor _actionExecutor;
 
    public async Task ProcessEventAsync(
        string triggerType, int projectId,
        Issue issue, CancellationToken ct)
    {
        var rules = await _db.AutomationRules
            .Where(r =&gt; r.ProjectId == projectId &amp;&amp; r.IsActive)
            .ToListAsync(ct);
 
        foreach (var rule in rules)
        {
            var trigger = rule.GetTriggerConfig();
            if (trigger.Type != triggerType) continue;
 
            // Evaluate conditions
            bool allMet = rule.GetConditions().All(c =&gt; EvaluateCondition(c, issue));
            if (!allMet) continue;
 
            // Execute actions
            foreach (var action in rule.GetActions())
                await _actionExecutor.ExecuteAsync(action, issue, ct);
 
            rule.LastRunAt = DateTime.UtcNow;
        }
        await _db.SaveChangesAsync(ct);
    }
 
    private bool EvaluateCondition(AutomationCondition c, Issue issue) =&gt;
        c.Type switch
        {
            &quot;FieldEquals&quot; =&gt; issue.GetFieldValue(c.Field)?.ToString() == c.Value,
            &quot;PriorityIs&quot;  =&gt; issue.Priority.ToString() == c.Value,
            &quot;IssueTypeIs&quot; =&gt; issue.IssueType.ToString() == c.Value,
            _             =&gt; true
        };
}

// src/features/boards/KanbanBoard.tsx
import { DragDropContext, Droppable, Draggable, DropResult } from &apos;react-beautiful-dnd&apos;;
import { useBoardQuery, useTransitionIssueMutation } from &apos;../../api/boardApi&apos;;
 
export default function KanbanBoard({ boardId }: { boardId: number }) {
    const { data: board, isLoading } = useBoardQuery(boardId);
    const transition = useTransitionIssueMutation();
 
    const onDragEnd = async (result: DropResult) =&gt; {
        const { draggableId, destination, source } = result;
        if (!destination || destination.droppableId === source.droppableId) return;
 
        const issueId  = parseInt(draggableId);
        const statusId = parseInt(destination.droppableId);
 
        // Optimistic update
        queryClient.setQueryData([&apos;board&apos;, boardId], (old: Board) =&gt; ({
            ...old,
            columns: old.columns.map(col =&gt; {
                if (col.statusId === parseInt(source.droppableId))
                    return { ...col, issues: col.issues.filter(i =&gt; i.id !== issueId) };
                if (col.statusId === statusId)
                    return { ...col, issues: [...col.issues, movedIssue] };
                return col;
            })
        }));
 
        try {
            await transition.mutateAsync({ issueId, targetStatusId: statusId });
        } catch {
            queryClient.invalidateQueries({ queryKey: [&apos;board&apos;, boardId] });
        }
    };
 
    if (isLoading) return &lt;BoardSkeleton /&gt;;
 
    return (
        &lt;DragDropContext onDragEnd={onDragEnd}&gt;
            &lt;div className=&quot;board-container&quot;&gt;
                {board.columns.map(col =&gt; (
                    &lt;BoardColumn key={col.id} column={col} /&gt;
                ))}
            &lt;/div&gt;
        &lt;/DragDropContext&gt;
    );
}

export function useBoardSignalR(boardId: number) {
    const connRef = useRef&lt;HubConnection | null&gt;(null);
    const queryClient = useQueryClient();
 
    useEffect(() =&gt; {
        const conn = new HubConnectionBuilder()
            .withUrl(&apos;/hubs/pms&apos;, {
                accessTokenFactory: () =&gt; localStorage.getItem(&apos;access_token&apos;) ?? &apos;&apos;
            })
            .withAutomaticReconnect({
                nextRetryDelayInMilliseconds: attempt =&gt;
                    Math.min(1000 * Math.pow(2, attempt), 30_000)
            })
            .configureLogging(LogLevel.Warning)
            .build();
 
        conn.on(&apos;IssueStatusChanged&apos;, (evt) =&gt; {
            queryClient.invalidateQueries({ queryKey: [&apos;board&apos;, boardId] });
        });
 
        conn.on(&apos;NotificationReceived&apos;, (notif) =&gt; {
            // Dispatch to Redux notification slice
        });
 
        conn.start()
            .then(() =&gt; conn.invoke(&apos;JoinBoard&apos;, boardId))
            .catch(console.error);
 
        connRef.current = conn;
 
        return () =&gt; {
            conn.invoke(&apos;LeaveBoard&apos;, boardId).finally(() =&gt; conn.stop());
        };
    }, [boardId]);
 
    return connRef;
}

// Application/Search/IqlParser.cs
public class IqlParser
{
    // Parses: project = DIG AND status IN (&quot;In Progress&quot;) ORDER BY priority DESC
    public IqlQuery Parse(string iql)
    {
        var tokens = Tokenize(iql);
        return new IqlQuery
        {
            Clauses  = ParseClauses(tokens),
            OrderBy  = ParseOrderBy(tokens),
            MaxResults = 50
        };
    }
 
    private List&lt;IqlClause&gt; ParseClauses(IEnumerable&lt;Token&gt; tokens)
    {
        var clauses = new List&lt;IqlClause&gt;();
        // tokenise → field, operator, value(s)
        // supports: = != &gt; &gt;= &lt; &lt;= IN NOT IN ~ !~
        // supports functions: currentUser(), openSprints(), startOfDay()
        return clauses;
    }
}
 
// Application/Search/IqlToElasticsearchTranslator.cs
public class IqlToElasticsearchTranslator
{
    public SearchRequest Translate(IqlQuery query, int tenantId)
    {
        var must = new List&lt;Query&gt;();
 
        // Always scope to tenant
        must.Add(new TermQuery(&quot;tenantId&quot;) { Value = tenantId });
 
        foreach (var clause in query.Clauses)
        {
            must.Add(clause.Operator switch
            {
                &quot;=&quot;  =&gt; new TermQuery(MapField(clause.Field))
                           { Value = ResolveValue(clause.Value) },
                &quot;IN&quot; =&gt; new TermsQuery { Field = MapField(clause.Field),
                           Terms = clause.Values.Select(ResolveValue) },
                &quot;~&quot;  =&gt; new MatchPhraseQuery(MapField(clause.Field))
                           { Query = clause.Value.ToString()! },
                &quot;&gt;=&quot;  =&gt; new DateRangeQuery(MapField(clause.Field))
                           { GreaterThanOrEqualTo = ParseDate(clause.Value) },
                _    =&gt; throw new IqlParseException($&quot;Unsupported operator: {clause.Operator}&quot;)
            });
        }
 
        return new SearchRequest
        {
            Query = new BoolQuery { Must = must },
            Sort  = BuildSort(query.OrderBy),
            Size  = query.MaxResults,
            From  = query.StartAt
        };
    }
}
